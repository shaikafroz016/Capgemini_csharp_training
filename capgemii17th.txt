    class con_ex
    {

        int[] arr;
        public con_ex(int[] arr)
        {
            this.arr = arr;
        }

       public int sumarray()
        {
            int sum = 0;
            for(int i=0;i<arr.Length;i++)
            {
                sum = sum + arr[i];
            }
            return sum;
        }

        public bool checkval(int val)
        {
            bool flag = false;
            for (int i = 0; i < arr.Length; i++)
            {
               if(arr[i]==val)
                {
                    flag = true;
                }
            }
            return flag;
        }

        public int[] addval()
        {
            int[] arr1 = new int[arr.Length];
            for (int i = 0; i < arr.Length; i++)
            {
                arr1[i] = arr[i] + 100;
            }
            return arr1;
        }
    }
